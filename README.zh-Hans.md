### Civitai AI жЁЎеһӢе№іеҸ°

```python
import requests

# и®ҫзҪ® API еҹәзЎҖ URL дҪҝз”Ё Xget
base_url = "https://xget.xi-xu.me/civitai"

# иҺ·еҸ–жЁЎеһӢдҝЎжҒҜ
def get_model_info(model_id):
    """иҺ·еҸ– Civitai жЁЎеһӢдҝЎжҒҜ"""
    url = f"{base_url}/api/v1/models/{model_id}"
    response = requests.get(url)
    return response.json()

# дёӢиҪҪжЁЎеһӢ
def download_model(model_version_id, output_path):
    """дёӢиҪҪ Civitai жЁЎеһӢж–Үд»¶"""
    download_url = f"{base_url}/api/download/models/{model_version_id}"

    print(f"жӯЈеңЁдёӢиҪҪжЁЎеһӢзүҲжң¬ {model_version_id}...")

    response = requests.get(download_url, stream=True)
    response.raise_for_status()

    with open(output_path, 'wb') as f:
        for chunk in response.iter_content(chunk_size=8192):
            f.write(chunk)

    print(f"жЁЎеһӢе·ІдёӢиҪҪеҲ°: {output_path}")

# дҪҝз”ЁзӨәдҫӢ
model_id = 7240  # зӨәдҫӢжЁЎеһӢ ID
model_info = get_model_info(model_id)
print(f"жЁЎеһӢеҗҚз§°: {model_info['name']}")

# дёӢиҪҪз¬¬дёҖдёӘжЁЎеһӢзүҲжң¬
if model_info['modelVersions']:
    version_id = model_info['modelVersions'][0]['id']
    download_model(version_id, f"model_{version_id}.safetensors")
```

### й…ҚзҪ®еҸӮж•°

дҪ еҸҜд»ҘйҖҡиҝҮдҝ®ж”№ `src/config/index.js` жқҘиҮӘе®ҡд№үй…ҚзҪ®пјҡ

```javascript

  TIMEOUT_SECONDS: 30, // иҜ·жұӮи¶…ж—¶ж—¶й—ҙпјҲз§’пјү
  MAX_RETRIES: 3, // жңҖеӨ§йҮҚиҜ•ж¬Ўж•°
  RETRY_DELAY_MS: 1000, // йҮҚиҜ•е»¶иҝҹж—¶й—ҙпјҲжҜ«з§’пјү
  CACHE_DURATION: 1800, // зј“еӯҳжҢҒз»ӯж—¶й—ҙпјҲ1800з§’ = 30еҲҶй’ҹпјү
  SECURITY: {
```

- **йЈҺйҷ©иҮӘжӢ…еҺҹеҲҷ**пјҡдҪҝз”ЁиҖ…еә”иҮӘиЎҢиҜ„дј°дҪҝз”ЁйЈҺйҷ©пјҢзЎ®дҝқе…¶дҪҝз”ЁиЎҢдёәеҗҲжі•еҗҲи§„пјҢдёҚдҫөзҠҜз¬¬дёүж–№жқғзӣҠпјҢдёҚеҫ—е°Ҷжң¬еӯҳеӮЁеә“з”ЁдәҺд»»дҪ•иҝқжі•гҖҒдҫөжқғгҖҒжҒ¶ж„ҸжҲ–дёҚеҪ“з”ЁйҖ”
- **з¬¬дёүж–№е№іеҸ°еҗҲи§„**пјҡдҪҝз”ЁиҖ…еә”йҒөе®Ҳзӣёе…іе№іеҸ°зҡ„жңҚеҠЎжқЎж¬ҫгҖҒAPI дҪҝз”Ёж”ҝзӯ–гҖҒйҖҹзҺҮйҷҗеҲ¶еҸҠзүҲжқғиҰҒжұӮпјҢйҒҝе…ҚеҜ№жәҗе№іеҸ°йҖ жҲҗиҝҮиҪҪжҲ–е№Іжү°гҖӮеҗ„е№іеҸ°еҜ№е…¶еҶ…е®№гҖҒжңҚеҠЎеҸҠж”ҝзӯ–жӢҘжңүжңҖз»Ҳи§ЈйҮҠжқғ
- **зҹҘиҜҶдә§жқғдҝқжҠӨ**пјҡйҖҡиҝҮжң¬еӯҳеӮЁеә“иҺ·еҸ–зҡ„еҶ…е®№еҸ—зӣёеә”зүҲжқғжі•дҝқжҠӨгҖӮдҪҝз”ЁиҖ…еә”йҒөе®Ҳзӣёе…іи®ёеҸҜеҚҸи®®гҖҒзүҲжқғеЈ°жҳҺеҸҠдҪҝз”ЁжқЎж¬ҫпјҢдёҚеҫ—д»ҺдәӢд»»дҪ•дҫөзҠҜзҹҘиҜҶдә§жқғзҡ„иЎҢдёә
- **е®үе…ЁйҳІжҠӨе»әи®®**пјҡиҷҪз„¶жң¬еӯҳеӮЁеә“йҮҮз”Ёж— ж—Ҙеҝ—жһ¶жһ„пјҢдёҚеӯҳеӮЁз”ЁжҲ·иҜ·жұӮж•°жҚ®пјҢдҪҶеҹәдәҺдә’иҒ”зҪ‘дј иҫ“зҡ„еӣәжңүйЈҺйҷ©пјҢе»әи®®дҪҝз”ЁиҖ…еҜ№дёӢиҪҪеҶ…е®№иҝӣиЎҢе®үе…Ёжү«жҸҸпјҢе°Өе…¶еҜ№еҸҜжү§иЎҢж–Үд»¶гҖҒи„ҡжң¬зӯүдҝқжҢҒи°Ёж…Һ
- **ејҖжәҗжҖ§иҙЁеЈ°жҳҺ**пјҡжң¬еӯҳеӮЁеә“дёәејҖжәҗйЎ№зӣ®пјҢдҪңиҖ…дёҺиҙЎзҢ®иҖ…дёҚжүҝжӢ…жҸҗдҫӣжҠҖжңҜж”ҜжҢҒгҖҒй”ҷиҜҜдҝ®еӨҚжҲ–жҢҒз»ӯз»ҙжҠӨзҡ„д№үеҠЎгҖӮеӨ–йғЁиҙЎзҢ®зҡ„еҗҲе№¶дёҚд»ЈиЎЁеҜ№зү№е®ҡз”ЁйҖ”жҲ–ж•Ҳжһңзҡ„жүҝиҜәдёҺиғҢд№Ұ
- **еҗҚз§°дҪҝз”Ёи§„иҢғ**пјҡдёҘзҰҒд»»дҪ•еҸҜиғҪжҡ—зӨәдҪңиҖ…жҲ–иҙЎзҢ®иҖ…жҸҗдҫӣе•ҶдёҡеҗҲдҪңгҖҒжҠҖжңҜж”ҜжҢҒгҖҒжӢ…дҝқжҲ–иғҢд№Ұзҡ„иЎЁиҝ°гҖӮж¶үеҸҠеӯҳеӮЁеә“еҗҚз§°жҲ–дҪңиҖ…ж ҮиҜҶзҡ„дҪҝз”Ёеә”йҒөеҫӘзӣёе…іжі•еҫӢжі•и§„еҸҠйҖҡз”Ёи§„иҢғ
- **е…ҚиҙЈеЈ°жҳҺжӣҙж–°**пјҡжң¬е…ҚиҙЈеЈ°жҳҺеҸҜиғҪйҡҸеӯҳеӮЁеә“еҸ‘еұ•жҲ–жі•еҫӢзҺҜеўғеҸҳеҢ–иҝӣиЎҢжӣҙж–°дҝ®и®ўгҖӮдҪҝз”ЁиҖ…з»§з»ӯдҪҝз”ЁгҖҒеӨҚеҲ¶гҖҒеҲҶеҸ‘жҲ–дҝ®ж”№жң¬еӯҳеӮЁеә“еҚіи§ҶдёәжҺҘеҸ—жңҖж–°зүҲжң¬зҡ„е…ҚиҙЈеЈ°жҳҺ
